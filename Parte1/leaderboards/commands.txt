CREATE TABLE user_monster_kills (
    country text,
    user_name text,
    event_id int,
    monster_id int,
    email text,
    PRIMARY KEY ((country, event_id), email)
);

CREATE TABLE user_kills_counter (
    country text,
    event_id int,
    email text,
    kills counter,
    PRIMARY KEY ((country, event_id), email)
);

CREATE TABLE best_players_by_country_and_dungeon (
    country text,
    dungeon_id int,
    time_minutes int,
    user_name text,
    email text,
    date,
    PRIMARY KEY ((country, dungeon_id), time_minutes, email)
) WITH CLUSTERING ORDER BY (time_minutes ASC);
